```java
class Solution {
    class Node{
        int x;
        int y;
    }
    public int numIslands(char[][] grid) {
        int n = grid.length; 
        if(n == 0) return 0;
        int m = grid[0].length;
        if(m == 0) return 0;
        boolean[][] visited = new boolean[n][m];
        Queue<Node> list = new LinkedList<Node>();
        int res = 0;
        int[] dx={1,0,-1,0};
        int[] dy={0,1,0,-1};
        for(int i = 0; i < n; i++){
            for (int j = 0; j < m; j++){
                if(!visited[i][j] && grid[i][j] == '1'){
                    Node t = new Node();
                    t.x = i;
                    t.y = j;
                    list.add(t);
                    System.out.println(t.x);
                    System.out.println(t.y);
                    visited[i][j] = true;
                }else continue;
                while(!list.isEmpty()){
                    Node t = list.poll();
                    for (int k = 0; k < 4; k++){
                        int x = t.x + dx[k];
                        int y = t.y + dy[k];
                        if(x >= 0 && x < n && y >= 0 && y < m) {
                            if(!visited[x][y] && grid[x][y] == '1'){
                                Node temp = new Node();
                                temp.x = x;
                                temp.y = y;
                                list.add(temp);
                                visited[x][y] = true;
                            }
                        }
                    }
                }
                res++;
            }
        }
        return res;
    }
}
```

```python
class Solution:
    row_num = 0
    col_num = 0
    def dfs(self, grid, row, col):
        if grid[row][col] == '1':
            grid[row][col] = 'X'
            if row > 0:
                self.dfs(grid, row-1, col)
            if col > 0:
                self.dfs(grid, row, col-1)
            if row < row_num-1:
                self.dfs(grid, row+1, col)
            if col < col_num-1:
                self.dfs(grid, row, col+1)
    def numIslands(self, grid: List[List[str]]) -> int:
        count = 0
        global row_num
        global col_num
        row_num = len(grid)
        col_num = len(grid[0])
        
        for i in range(row_num):
            for j in range(col_num):
                if grid[i][j] != '1':
                    continue
                self.dfs(grid, i, j)
                count += 1
        return count
```

```python
class Solution:
    row_num = 0
    col_num = 0
    def dfs(self, grid, row, col):
        if grid[row][col] == '1':
            grid[row][col] = 'X'
            if row > 0:
                self.dfs(grid, row-1, col)
            if col > 0:
                self.dfs(grid, row, col-1)
            if row < row_num-1:
                self.dfs(grid, row+1, col)
            if col < col_num-1:
                self.dfs(grid, row, col+1)
    def numIslands(self, grid: List[List[str]]) -> int:
        count = 0
        n = len(grid)
        m = len(grid[0])
        
        directions = [(1, 0), (-1, 0), (0, 1), (0, -1)]
        queue = deque()
        for i in range(n):
            for j in range(m):
                if grid[i][j] != '1':
                    continue
                grid[i][j] = '0'
                queue.append((i,j))
                while queue:
                    x, y = queue.popleft()
                    for dx, dy in directions:
                        new_x = x + dx
                        new_y = y + dy
                        if (new_x < 0 or new_x > n-1 or new_y < 0 or new_y > m-1 or grid[new_x][new_y] == '0'):
                            continue
                        grid[new_x][new_y] = '0'
                        queue.append((new_x, new_y))
                count += 1
        return count
```

```python
class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        m, n, dict1 = len(grid), len(grid[0]), defaultdict(int)
        def find(x):
            if x not in dict1:
                return x
            else:
                if x != dict1[x]:
                    dict1[x] = find(dict1[x])
                return dict1[x]
        def union(x, y):
            a, b = find(x), find(y)
            if a != b:
                dict1[a] = b
        
        for i in range(m):
            for j in range(n):
                if grid[i][j] == '1' and (i+1 < m and grid[i+1][j] == '1'):
                    union(i*n+j, (i+1)*n+j)
                if grid[i][j] == '1' and (i > 0 and grid[i-1][j] == '1'):
                    union(i*n+j, (i-1)*n+j)
                if grid[i][j] == '1' and (j+1 < n and grid[i][j+1] == '1'):
                    union(i*n+j, i*n+j+1)
                if grid[i][j] == '1' and (j > 0 and grid[i][j-1] == '1'):
                    union(i*n+j, i*n+j-1)
        res_set = set()
        for i in range(m):
            for j in range(n):
                if grid[i][j] == '1':
                    res_set.add(find(i*n+j))
        return len(res_set)
```
