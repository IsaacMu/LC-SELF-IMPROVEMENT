# 139. Word Break

## Solution

```python
class Solution:
    def wordBreak(self, s: str, wordDict: List[str]) -> bool:
        words = set(wordDict)
        queue = collections.deque([0])
        seen = set()
        while queue:
            start = queue.popleft()
            if start == len(s):
                return True
    
            for end in range(start+1, len(s) +1):
                if end in seen:
                    continue
                if s[start:end] in words:
                    queue.append(end)
                    seen.add(end)
        return False
```
Time complexity is O(n3) , Space complexity is O(n)

### DP method
```python
class Solution:
    def wordBreak(self, s: str, wordDict: List[str]) -> bool:
        dp = [False for i in range(len(s)+1)]
        queue = collections.deque([0])
        visited = set()
        while queue:
            index = queue.popleft()
            for word in wordDict:
                if s[index:].startswith(word):
                    new_index = index+len(word)
                    if new_index in visited:
                        continue
                    dp[new_index] = True
                    queue.append(new_index)
                    visited.add(new_index)
        return dp[len(s)]
```
### Trie
```python
class TrieNode:
    def __init__(self):
        self.is_word = False
        self.children = {}
class Solution:
    def wordBreak(self, s: str, wordDict: List[str]) -> bool:
        root = TrieNode()
        for word in wordDict:
            cur = root
            for c in word:
                if c not in cur.children:
                    cur.children[c] = TrieNode()
                cur = cur.children[c]
            cur.is_word = True

        dp = [False for i in range(len(s))]
        for i in range(len(s)):
            if i == 0 or dp[i-1]:
                cur = root
                for j in range(i, len(s)):
                    c = s[j]
                    if c not in cur.children:
                        break
                    cur = cur.children[c]
                    if cur.is_word:
                        dp[j] = True
        return dp[-1]
```
